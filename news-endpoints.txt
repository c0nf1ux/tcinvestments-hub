app.get('/api/news/:id', (req, res) => {
  // Individual news article endpoint
  const { id } = req.params;
  const article = mockNews.find(n => n.id === parseInt(id));
  
  if (!article) {
    return res.status(404).json({ message: 'Article not found' });
  }
  
  res.json(article);
});
